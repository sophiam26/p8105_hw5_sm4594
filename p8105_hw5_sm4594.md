``` r
replace_missing = function(x) {
  
   if (is.numeric(x)) {
    ifelse(is.na(x), mean(x, na.rm = TRUE), x)
   } else if (is.character(x)) {
     ifelse(is.na(x), "virginica", x)
   }
     
}

iris_complete = map_df(iris_with_missing, replace_missing)
```
